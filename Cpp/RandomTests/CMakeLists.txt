function(add_code name)
    add_executable(rt_${name} ${name}.cpp)
    target_include_directories(rt_${name} PRIVATE
                               ${Ecosystem_SOURCE_DIR}/include)
    target_link_libraries(rt_${name} PRIVATE ${ECOSYSTEM_LIB} dl fmt::fmt pthread)
endfunction()
